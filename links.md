	function secondToDate(second) {
		if (!second) {
		return 0;
		}
		var time = new Array(0, 0, 0, 0, 0);
		if (second >= 365 * 24 * 3600) {
		 time[0] = parseInt(second / (365 * 24 * 3600));
		 second %= 365 * 24 * 3600;
		}
		if (second >= 24 * 3600) {
		 time[1] = parseInt(second / (24 * 3600));
		 second %= 24 * 3600;
		}
		if (second >= 3600) {
		 time[2] = parseInt(second / 3600);
		 second %= 3600;
		}
		if (second >= 60) {
		 time[3] = parseInt(second / 60);
		 second %= 60;
		}
		if (second > 0) {
		 time[4] = second;
		}
		return time;
	}
